% ══ ROW: PROBLEM (col 1) | APPROACH (col 2) ══

\posterbox[
  redcard, title={\bfseries Problem We Tackle}
]{name=problem, column=1, between=row7 and row14}{
  \fontsize{28}{38}\selectfont
  \textbf{Spurious correlation} 환경에서 ERM 은 majority 의 \emph{지름길 신호}
  (e.g.\ Colored MNIST 의 색)를 잡아 minority / OOD worst-group accuracy 를
  무너뜨린다.\\[6pt]

  \textbf{표준 해법의 한계.}
  \begin{itemize}[itemsep=4pt]
    \item Bi-level meta-learning (Ren 2018, FORML): minibatch 마다 inner SGD
          + implicit differentiation \textemdash{} 구현 무겁고 비용 큼.
    \item GroupDRO / IRM: \textbf{group label} 이 학습 중 필요.
    \item JTT: 2-stage ERM, \emph{학습 동역학} 으로의 개입은 없음.
  \end{itemize}
  \vspace{2pt}
  \colorbox{secondary!20}{\parbox{0.94\linewidth}{\fontsize{26}{34}\selectfont
    \textbf{Wang et al. 2024} 는 \emph{closed-form} 1차 / 2차 In-Run
    Shapley estimator 를 단일 학습 run 으로 유도했다 \textemdash{} 그러나
    \textbf{post-hoc filtering} (저-Shapley 샘플 제거 후 재학습) 에만 사용했다.}}
  \vspace{6pt}

  \textbf{우리의 질문:} 같은 closed-form 수식을 \emph{학습 중 per-sample
  reweighter} 로 재용도(repurpose) 할 수 있는가? 그리고 2차 cross-term 이
  1차 단독 대비 \textbf{stand-alone 기여}를 주는가?
}

\posterbox[
  bluecard, title={\bfseries How We Approach It}
]{name=approach, column=2, between=row7 and row14}{
  \fontsize{28}{38}\selectfont
  \textbf{핵심 아이디어 1 — 사후 필터링을 학습 중 가중치로.}\\
  Shapley 값을 학습이 끝난 뒤 사용하는 대신, 매 minibatch 마다
  per-sample Shapley 추정치를 softmax 가중치로 환산해 가중 SGD 수행.\\[6pt]

  \textbf{핵심 아이디어 2 — 2차 cross-term 의 기하학적 의미.}\\
  1차 항 $\langle g_i, g_{\mathrm{val}}\rangle$ 은 단순 gradient alignment.\\
  2차 cross-term $\langle g_i, H_{\mathrm{val}}\, g_{\mathrm{val}}\rangle$ 은
  \emph{validation loss 의 곡률이 큰 방향에 align 된 majority gradient} 에
  추가 페널티 \textemdash{} curvature-aware down-weighting.\\[4pt]

  \colorbox{accent!18}{\parbox{0.94\linewidth}{\fontsize{26}{34}\selectfont
    \textbf{핵심 아이디어 3 — RMS rescaling.}\\
    학습 중 Hessian spectrum 이 수십 배 변동 \textemdash{} cross-term 을
    그대로 더하면 발산. 매 step 1차 항의 RMS 에 맞춰 재스케일.}}
  \vspace{6pt}

  \textbf{핵심 아이디어 4 — EMA 누적.}\\
  Per-step $\phi_i$ 는 noisy. iteration 간 EMA 로 평활화 후 softmax-weight.
  \vspace{2pt}
  \begin{itemize}[itemsep=4pt]
    \item No implicit differentiation \textemdash{} 단일 forward + backward + HVP.
    \item Group label 은 anchor 50:50 균형에만 사용 (학습 중 불필요).
  \end{itemize}
}

% ══ METHOD (span 2 columns) ══

\posterbox[
  darkcard, title={\bfseries Method: Fairds — 5-Step In-Training Pipeline}
]{name=method, column=1, span=2, between=row14 and row24}{
  \fontsize{26}{34}\selectfont
  파라미터 $\theta_t$, minibatch $\mathcal{B}_t = \{z_i\}_{i=1}^B$, anchor
  $\mathcal{D}_{\mathrm{val}}$ (50:50 group-balanced, $|\mathcal{D}_{\mathrm{val}}|\!=\!200$). 매 step:
  \vspace{6pt}

  \begin{minipage}[t]{0.58\linewidth}
    \textbf{\stepbadge{1}\ Per-sample gradient $g_i$.}\\
    {\fontsize{24}{32}\selectfont $g_i = \nabla_\theta \mathcal{L}(z_i;\theta_t)$, \texttt{torch.func.vmap(grad)} 으로 단일 autograd graph 에서 한 번에.}
    \\[10pt]

    \textbf{\stepbadge{2}\ Validation gradient \& HVP.}\\
    {\fontsize{24}{32}\selectfont
    $g_{\mathrm{val}}\!=\!\nabla_\theta \mathcal{L}(\mathcal{D}_{\mathrm{val}};\theta_t)$,\quad
    $H_{\mathrm{val}}\,g_{\mathrm{val}}$ via Pearlmutter HVP \textemdash{} full Hessian 불필요, $\mathcal{O}(P)$ 메모리.}
    \\[10pt]

    \textbf{\stepbadge{3}\ Closed-form Shapley score.}\\
    \colorbox{accent!20}{\parbox{0.92\linewidth}{\fontsize{24}{32}\selectfont\centering\vspace{3pt}
      $\displaystyle \phi_i^{(1)} = \langle g_i,\, g_{\mathrm{val}}\rangle$\hfill\textit{(1차)}\\[2pt]
      $\displaystyle \phi_i^{(2)} = \phi_i^{(1)} \;-\; \alpha\,\widetilde{c}_i,\quad c_i\!=\!\langle g_i,\, H_{\mathrm{val}}\, g_{\mathrm{val}}\rangle$\hfill\textit{(2차)}\\[2pt]
      $\displaystyle \widetilde{c}_i = c_i \cdot \tfrac{\mathrm{RMS}_i(\phi_i^{(1)})}{\mathrm{RMS}_i(c_i) + \epsilon}$\hfill\textit{(RMS rescale)}\vspace{3pt}}}
    \\[8pt]

    \textbf{\stepbadge{4}\ EMA accumulate.}\\
    {\fontsize{24}{32}\selectfont
    $\overline{\phi}_i \leftarrow \beta\,\overline{\phi}_i + (1-\beta)\,\phi_i^{(2)}$,\quad $\beta\!=\!0.9$.\\
    Per-step $\phi$ 의 분산을 줄이고 minibatch boundary 효과 완화.}
    \\[10pt]

    \textbf{\stepbadge{5}\ Softmax weight \& weighted SGD.}\\
    {\fontsize{24}{32}\selectfont
    $\displaystyle w_i = \mathrm{clip}_{q_{0.05},q_{0.95}}\!\Bigl(\tfrac{B \cdot \exp(\overline{\phi}_i / \tau)}{\sum_j \exp(\overline{\phi}_j / \tau)}\Bigr)$,\quad
    $\theta_{t+1} = \theta_t - \eta \sum_i w_i\,g_i$.}
  \end{minipage}\hfill
  \begin{minipage}[t]{0.39\linewidth}
    \colorbox{secondary!18}{\parbox{0.94\linewidth}{\fontsize{24}{30}\selectfont\vspace{4pt}
      \textbf{Hyperparameters}\\[3pt]
      \begin{tabular}{ll}
        $\alpha$ & 2차 weight $=1.0$\\
        $\tau$ & softmax temp $=0.5$\\
        $\beta$ & EMA decay $=0.9$\\
        quantile clip & $[0.05,\,0.95]$\\
        $|\mathcal{D}_{\mathrm{val}}|$ & $200$ (50:50)\\
      \end{tabular}\vspace{4pt}}}
    \\[10pt]
    \colorbox{primary!18}{\parbox{0.94\linewidth}{\fontsize{24}{30}\selectfont\vspace{4pt}
      \textbf{Variants compared}\\[3pt]
      \textbf{Fairds-1}: 식에서 $\alpha\!=\!0$ (1차만).\\
      \textbf{Fairds-2}: $\alpha\!=\!1$ + RMS rescale.\\[4pt]
      두 variant 의 차이가 \emph{2차 cross-term 의 isolated 기여}.\vspace{4pt}}}
    \\[10pt]
    \colorbox{accent!18}{\parbox{0.94\linewidth}{\fontsize{22}{30}\selectfont\vspace{4pt}
      \textbf{비용 (per step)}\\[3pt]
      \begin{itemize}[leftmargin=14pt, itemsep=2pt]
        \item $g_i$: vmap, 1 fwd + 1 bwd
        \item $g_{\mathrm{val}}$: 1 fwd + 1 bwd
        \item HVP: 1 추가 bwd (Pearlmutter)
      \end{itemize}
      Wall-clock: vanilla 의 $5{-}7\!\times$ (small CNN).\vspace{4pt}}}
  \end{minipage}
  \vspace{4pt}

  \colorbox{secondary!20}{\parbox{0.97\linewidth}{\fontsize{26}{34}\selectfont\centering\vspace{4pt}
    \textbf{알고리즘적 기여:} cross-term 의 RMS rescaling \textemdash{} 빠지면 strong spurious bias 에서 Fairds-2 가 chance 까지 붕괴. ablation 으로 확인.\vspace{4pt}}}
}

% ══ TAKEAWAYS (span 2) ══

\posterbox[
  highlightcard, title={\bfseries Takeaways}
]{name=takeaways, column=1, span=2, between=row33 and row36}{
  \fontsize{28}{38}\selectfont\color{textdark}
  \begin{enumerate}[itemsep=4pt]
    \item Closed-form 2nd-order Shapley reweighting 은 \textbf{regime-dependent} 도구다.
    \item \textbf{From-scratch / spurious-correlation} 환경에서 작동 \textemdash{} 2차 cross-term 의 기여를 통계적으로 isolate 했다 ($+4.7$\,pp, $p\!=\!0.033$).
    \item \textbf{Pretrained fine-tuning} 환경에서는 실패 \textemdash{} bi-level meta-learning 이 더 강하다 (정직 보고).
    \item \textbf{Cross-term 의 RMS rescaling} 이 안정성의 키 (없으면 발산).
  \end{enumerate}
}
